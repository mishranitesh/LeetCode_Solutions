Given an integer, write a function to determine if it is a power of two.

1           true
16          true
218         false
-1          false
INT_MIN     false

-------------------------------------------------------------------------------------------------------------

class Solution {
public:
    bool isPowerOfTwo(int n) {
        
        /*
         * Efficient Approach -- Using bit-fiddling
         * [x & (x - 1)]  --> Removes last set bit from LSB
         * [x & ~(x - 1)] --> Isolates last set bit from LSB
         */
        return (n > 0) && ((n & (n - 1)) == 0);
        
    }
};
